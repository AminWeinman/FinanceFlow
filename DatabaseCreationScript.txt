CREATE DATABASE pff;

USE pff;

CREATE TABLE users (
uid INT PRIMARY KEY AUTO_INCREMENT,
name VARCHAR(50),
pwd VARCHAR(50)
);


create table transactions (
tid int primary key auto_increment,
uid int not null,
transactions_type enum('income', 'expenses') not null,
amt double,
description TEXT,
date datetime, 
foreign key (uid) references users(uid)
);
